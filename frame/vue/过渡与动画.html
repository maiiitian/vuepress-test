# Transition

<Transition> 会在一个元素或组件进入和离开 DOM 时应用动画。

Hello1.vue

<template>
  <div>
    <button @click="isShow = !isShow">显示/隐藏</button>
    <!-- appear刷新页面就出现渐入效果 -->
    <transition appear>
        <h1 v-show="isShow">啊啊啊啊</h1>
    </transition>
    <transition name="hello" appear>
        <h1 v-show="isShow">你好啊！</h1>
    </transition>
  </div>
</template>

<script>
export default {
    name: 'Hello1',
    data(){
        return{
            isShow: true
        }
    }
}
</script>

<style>
    h1{
        background-color: orange ;
    }
    /*进入过程*/
    .v-enter-active{
        animation: identifier 0.5s linear;
    }
    /*离开过程*/
    .v-leave-active{
        animation: identifier 0.5s linear reverse;
    }
    /* 默认是v，如果在transition元素name属性中命名，则使用改命名为前缀 */
    .hello-enter-active{
        animation: identifier 0.5s linear;
    }
    .hello-leave-active{
        animation: identifier 0.5s linear reverse;
    }
    @keyframes identifier {
        from{
            transform: translateX(-100%);
        }
        to{
            transform: translateX(0px);
        }
    }

    
</style>

# TransitionGroup

<TransitionGroup> 会在一个 v-for 列表中的元素或组件被插入，移动，或移除时应用动画。

Hello2.vue

<template>
    <div>
        <button @click="isShow = !isShow">显示/隐藏</button>
        <!-- <transition name="hi" appear>
            <div>
                <h1 v-show="isShow">你好啊啊啊啊！</h1>
                <h1 v-show="isShow">你好啊啊啊啊！</h1>
            </div>
        </transition> -->
        <transition-group name="hi" appear>
            <h1 v-show="isShow" key="1">你好啊啊啊啊！</h1>
            <h1 v-show="!isShow" key="2">你好啊啊啊啊！</h1>
        </transition-group>
    </div>
</template>
  
<script>
export default {
    name: 'Hello2',
    data() {
        return {
            isShow: true
        }
    }
}
</script>
  
<style>
h1 {
    background-color: orange;
}
/*进入的起点，离开的终点*/
.hi-enter,
.hi-leave-to {
    transform: translateX(-100%);
}

.hi-enter-active,
.hi-leave-active {
    transition: 0.5s linear;
}
/*进入的终点，离开的起点*/
.hi-enter-to,
.hi-leave {
    transform: translateX(0px);
}
</style>

# 第三方动画 animate.css

Hello3.vue

<template>
    <div>
        <button @click="isShow = !isShow">显示/隐藏</button>
        <!-- appear刷新页面就出现渐入效果 -->
        <transition-group appear name="animate__animated animate__bounce" enter-active-class="animate__swing"
            leave-active-class="animate__backOutUp">
            <h1 v-show="!isShow" key="1">你好啊！</h1>
            <h1 v-show="isShow" key="2">菜狗！</h1>
        </transition-group>
    </div>
</template>
  
<script>
import 'animate.css'
export default {
    name: 'Hello1',
    data() {
        return {
            isShow: true
        }
    }
}
</script>
  
<style>
h1 {
    background-color: orange;
}
</style>
